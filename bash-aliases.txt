slice() {
  git submodule foreach "git add ."
  git submodule foreach "git commit -m 'Slice'"

  git add -A
  git commit -m "Slice"

  git submodule foreach "git add ."
  git submodule foreach "git commit -m 'Slice'"
}

commit() {  
  if [ -n "$1" ]; then
    git submodule foreach "git add ."
    git submodule foreach "git commit -m ""$1"  

    git add -A
    git commit -m "$1"
  else
    git submodule foreach "git add ."
    git submodule foreach "git commit -m 'Slice'"

    git add -A
    git commit -m "Slice"
  fi
}
